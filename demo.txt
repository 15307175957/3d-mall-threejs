        var floorControl;
        var building;
        //楼层控制控件配置参数
        var ctlOpt = new esmap.ESControlOptions({
            position: esmap.ESControlPositon.RIGHT_TOP,
            imgURL: "image/wedgets/"
        });
        var container = document.getElementById('map-container');
        var map = new esmap.ESMap({
            mode: esmap.MapMode.Building,
            container: container, //渲染dom
            mapDataSrc: '../data', //三维场景数据位置
            mapAudioSrc: '../lib',
	        token:'escope',
            visibleFloors: "all"
        });
        //打开三维场景数据
        map.openMapById(esmapID);
  
        map.on("loadComplete", function () {
            building = map.getBuilding();
            document.getElementById("focus").value = building.focusFloorNum
            //创建楼层控件
            floorControl = new esmap.ESScrollFloorsControl(map, ctlOpt);
            building.on("focusFloorNumChanged", function (e) {
            document.getElementById("focus").value = e
            })
        })
        map.showCompass = true; //显示指南针
